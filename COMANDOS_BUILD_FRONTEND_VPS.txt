# COMANDOS PARA FAZER BUILD DO FRONTEND NA VPS
# ==============================================

# IMPORTANTE: Após corrigir código TypeScript/JavaScript no frontend,
# você DEVE fazer build para gerar os arquivos JavaScript compilados.
# Os arquivos .ts/.tsx NÃO são usados diretamente no navegador!

cd /root/primecamp-ofc

# 1. Atualizar código do Git
git pull origin main

# 2. Instalar dependências (se necessário)
npm install

# 3. Fazer build do frontend
npm run build

# 4. Copiar dist para TODAS as pastas que o Nginx usa (grep root em /etc/nginx/)
#    primecamp.cloud → /var/www/primecamp.cloud | ativafix.com pode → /var/www/primecamp.cloud
#    Outros hosts → /var/www/ativafix
for WWW in /var/www/ativafix /var/www/primecamp.cloud; do
  if [ -d "$WWW" ]; then
    sudo rm -rf "$WWW"/*
    sudo cp -r dist/* "$WWW"/
    sudo chown -R www-data:www-data "$WWW"
  fi
done

# 5. Recarregar Nginx
sudo systemctl reload nginx
